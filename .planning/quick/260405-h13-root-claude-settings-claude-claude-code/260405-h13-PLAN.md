在 admin_hosts.go 中新增以下 handler 方法，全部挂在 AdminHostsHandler 上，复用现有的 host 查找和 running 状态检查模式：

1. **ChangeRootPassword()** — `POST /v1/admin/hosts/{hostID}/change-root-password`
   - 请求体：`{ "password": "新密码" }`，password 不能为空
   - 获取 host，检查 status == "running"，否则返回 409
   - 使用 `syncContainerPassword(containerName, "root", password)` 修改 root 密码（复用已有函数）
   - 记录事件 `admin.host.root_password_changed`
   - 返回 `{ "status": "ok" }`

2. **GetClaudeSettings()** — `GET /v1/admin/hosts/{hostID}/claude/settings`
   - 获取 host，检查 running
   - `docker exec -i containerName cat /workspace/.claude/settings.json`
   - 如果文件不存在，返回空对象 `{}`
   - 将 stdout 作为 raw JSON 返回：`{ "settings": <parsed JSON> }`

3. **UpdateClaudeSettings()** — `PUT /v1/admin/hosts/{hostID}/claude/settings`
   - 请求体：`{ "settings": { ... } }`，整个 settings 对象
   - 获取 host，检查 running
   - 先 `docker exec containerName mkdir -p /workspace/.claude`
   - 将 settings JSON 序列化后通过 stdin 写入：`docker exec -i containerName tee /workspace/.claude/settings.json`
   - 记录事件 `admin.host.claude_settings_updated`
   - 返回 `{ "status": "ok" }`

4. **GetClaudeStatus()** — `GET /v1/admin/hosts/{hostID}/claude/status`
   - 获取 host，检查 running
   - 执行 `docker exec containerName bash -c 'pgrep -c -f claude-real 2>/dev/null || echo 0'` 获取进程数
   - 执行 `docker exec containerName bash -c 'claude-real --version 2>/dev/null || echo unknown'` 获取版本
   - 返回 `{ "running_instances": <int>, "version": "<string>" }`

5. **UpdateClaude()** — `POST /v1/admin/hosts/{hostID}/claude/update`
   - 获取 host，检查 running
   - 执行 `docker exec containerName bash -c 'npm install -g @anthropic-ai/claude-code@latest 2>&1 && NEWPATH=$(npm root -g)/@anthropic-ai/claude-code/cli.js && cp "$NEWPATH" /usr/local/bin/claude-real 2>&1'`
   - 使用 30 秒超时（npm install 可能较慢）
   - 更新完成后获取新版本号
   - 记录事件 `admin.host.claude_updated`
   - 返回 `{ "status": "ok", "version": "<新版本>" }`

**A) use-hosts.ts — 新增 5 个 hooks：**

```typescript
// 1. 修改 root 密码
export function useChangeRootPassword() {
  const qc = useQueryClient();
  return useMutation({
    mutationFn: ({ hostId, password }: { hostId: string; password: string }) =>
      apiFetch(`/hosts/${hostId}/change-root-password`, {
        method: "POST",
        body: JSON.stringify({ password }),
      }),
    onSuccess: (_data, { hostId }) => {
      qc.invalidateQueries({ queryKey: ["hosts", hostId] });
    },
  });
}

// 2. 读取 Claude settings
export function useClaudeSettings(hostId: string, enabled = true) {
  return useQuery({
    queryKey: ["hosts", hostId, "claude-settings"],
    queryFn: () => apiFetch<{ settings: Record<string, unknown> }>(`/hosts/${hostId}/claude/settings`),
    enabled: !!hostId && enabled,
  });
}

// 3. 更新 Claude settings
export function useUpdateClaudeSettings() {
  const qc = useQueryClient();
  return useMutation({
    mutationFn: ({ hostId, settings }: { hostId: string; settings: Record<string, unknown> }) =>
      apiFetch(`/hosts/${hostId}/claude/settings`, {
        method: "PUT",
        body: JSON.stringify({ settings }),
      }),
    onSuccess: (_data, { hostId }) => {
      qc.invalidateQueries({ queryKey: ["hosts", hostId, "claude-settings"] });
    },
  });
}

// 4. Claude 运行状态
export function useClaudeStatus(hostId: string, enabled = true) {
  return useQuery({
    queryKey: ["hosts", hostId, "claude-status"],
    queryFn: () => apiFetch<{ running_instances: number; version: string }>(`/hosts/${hostId}/claude/status`),
    enabled: !!hostId && enabled,
    refetchInterval: 30000, // 每 30 秒自动刷新
  });
}

// 5. 更新 Claude Code
export function useUpdateClaude() {
  const qc = useQueryClient();
  return useMutation({
    mutationFn: (hostId: string) =>
      apiFetch<{ status: string; version: string }>(`/hosts/${hostId}/claude/update`, {
        method: "POST",
      }),
    onSuccess: (_data, hostId) => {
      qc.invalidateQueries({ queryKey: ["hosts", hostId, "claude-status"] });
    },
  });
}
```
